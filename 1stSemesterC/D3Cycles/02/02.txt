#include <stdio.h>

int main()
{
	/*2. Пирамида от числа
    Да се отпечатат числата 1 … n в пирамида като в примерите по долу.
	На първия ред печатаме едно число, на втория ред печатаме две 
	числа, на третия ред печатаме три числа и т.н. докато числата 
	свършат. На последния ред печатаме толкова числа, колкото останат 
	докато стигнем до n.
    */
	int input;

	// Вход - големиан на пирадимата
	printf("Enter the value of input: ");

	// Четене на стойността на входа от потребителя
	scanf("%d", &input);

	int currentNumber = 1;

	for (int i = 1; currentNumber <= input; i++)
	{
		// Външен цикъл за редовете
		for (int j = 0; j < i && currentNumber <= input; j++)
		{
			// Вътрешен цикъл за числата на всеки ред
			printf("%d ", currentNumber); // Отпечатваме текущото число
			currentNumber++; // Увеличаваме текущото число с 1
		}
		printf("\n"); // Преминаваме на нов ред след всеки ред от пирамидата
	}


}
